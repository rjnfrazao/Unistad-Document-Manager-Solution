#Generate secure string password - change the password here
$pwd = ConvertTo-SecureString "tellno1" -AsPlainText -Force

#Generate Root CA - You can change the Root Certificate Name
$rootCa = GenerateRootCA -Password $pwd -RootCaName "Dev Root CA" -PfxPath "DevRootCA.pfx" -CrtPath "DevRootCA.cer" -Import $true

#Generate Certificate for localhost
GenerateCASignedCert -RootCAPath $rootCa -FriendlyName "localhost" -DomainDns "localhost,*.localhost.com" -Password $pwd -Import $true -CrtPath "localhost.cer" -PfxPath "localhost.pfx"
